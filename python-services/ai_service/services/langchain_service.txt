# python-services/ai_service/services/langchain_service.py

import os
from typing import List, Dict, Optional, Any
from langchain_google_genai import ChatGoogleGenerativeAI
from langchain_core.prompts import ChatPromptTemplate, MessagesPlaceholder
from langchain_core.messages import HumanMessage, AIMessage, SystemMessage
# from langchain_community.callbacks import get_openai_callback
from langchain_core.output_parsers import StrOutputParser
from config.settings import settings
import logging

logger = logging.getLogger(__name__)

class LangChainService:
    """Service for LangChain-based LLM interactions"""   
    
    def __init__(self):
        # Initialize LLM  
        self.llm = ChatGoogleGenerativeAI(
            model=settings.LLM_MODEL,
            temperature=settings.LLM_TEMPERATURE,
            max_output_tokens=settings.LLM_MAX_TOKENS,
            google_api_key=settings.GOOGLE_API_KEY
        )
        
        # Initialize Output Parser
        self.output_parser = StrOutputParser()
        
        # System prompt for mental health support
        self.system_prompt = self._load_system_prompt()

# 0. => a.  ==================================================================================================== #  
    def _load_system_prompt(self) -> str:
        """Load the mental health support system prompt"""
        return """You are a compassionate and empathetic AI mental health support assistant designed to help college students.

            Your role is to:
            1. Provide emotional support and validation
            2. Offer evidence-based coping strategies (CBT, mindfulness, grounding techniques)
            3. Encourage help-seeking behavior when appropriate
            4. Never diagnose or prescribe medication
            5. Maintain a warm, non-judgmental tone

            Important guidelines:
            - Always acknowledge the user's feelings
            - Use simple, clear language
            - Offer practical, actionable suggestions
            - If detecting severe distress or crisis keywords, gently encourage professional help
            - Never claim to replace professional mental health care
            - Respect confidentiality and privacy

            You have access to verified mental health resources through the RAG system.
            Use them to provide accurate, evidence-based information.

            Remember: You're here to support, not to diagnose or treat."""


# 1. ======================================================================================================================================= #  
    async def generate_response(
        self,
        user_message: str,
        conversation_history: List[Dict[str, Any]] = None,
        rag_context: Optional[List[Dict[str, Any]]] = None,
        safety_context: Optional[Dict[str, Any]] = None
    ) -> Dict[str, Any]:
        """
        Generate AI response using LangChain with RAG context
        """
        try:
            # Build messages list
            messages = [SystemMessage(content=self.system_prompt)]
            
            # Add conversation history
            if conversation_history:
                for msg in conversation_history[-6:]:  # Last 6 messages for context
                    if msg.get("role") == "user":
                        messages.append(HumanMessage(content=msg["content"]))
                    elif msg.get("role") == "assistant":
                        messages.append(AIMessage(content=msg["content"]))
            
            # Construct user message with RAG context
            enhanced_message = user_message
            
            if rag_context and len(rag_context) > 0:
                context_text = "\n\n".join([
                    f"Resource: {doc.get('title', 'Untitled')}\n{doc.get('content', '')}"
                    for doc in rag_context[:3]
                ])
                
                enhanced_message = f"""Based on the following verified mental health resources: 
                    {context_text}
                    User query: {user_message}
                    Please provide a helpful, empathetic response that incorporates relevant information from these resources when appropriate."""
            
            # Add safety context if high risk
            if safety_context and safety_context.get("risk_level") in ["high", "critical"]:
                enhanced_message += "\n\n[SYSTEM NOTE: User may be in distress. Provide gentle support and encourage seeking professional help.]"
            
            messages.append(HumanMessage(content=enhanced_message))
            
            # --- FIX STARTS HERE ---
            # Use 'invoke' instead of calling the object directly
            response_content = await self.llm.ainvoke(messages)
            
            # Extract content from AIMessage
            final_text = response_content.content
            # --- FIX ENDS HERE ---
            
            # Post-process response for safety
            final_response = self._post_process_response(
                final_text,
                safety_context
            )
            
            return {
                "message": final_response,
                "model": settings.LLM_MODEL,
                "tokens_used": 0 # Token counting is tricky with async Google models, setting to 0 for now to prevent errors
            }
            
        except Exception as e:
            logger.error(f"LangChain response generation error: {str(e)}")
            
            # Fallback response
            return {
                "message": self._get_fallback_response(safety_context),
                "model": "fallback",
                "tokens_used": 0,
                "error": str(e)
            }
    

# 1. => a.  ==================================================================================================== #  
    def _post_process_response(
        self,
        response: str,
        safety_context: Optional[Dict[str, Any]]
    ) -> str:
        """
        Post-process the LLM response for safety and appropriateness
        """
        # Add crisis resources if high risk
        if safety_context and safety_context.get("risk_level") in ["high", "critical"]:
            response += "\n\n⚠️ If you're in immediate danger or experiencing a crisis, please reach out to:\n"
            response += "- Emergency Helpline: 1-800-273-8255\n"
            response += "- Campus Counseling Services\n"
            response += "- Emergency Services: 911\n"
            response += "\nYou don't have to go through this alone. Help is available."
        
        # Ensure response doesn't make medical claims
        medical_disclaimers = [
            "I'm diagnosing",
            "I can diagnose",
            "You have a disorder",
            "You need medication",
            "Take this medication"
        ]
        
        for disclaimer in medical_disclaimers:
            if disclaimer.lower() in response.lower():
                logger.warning(f"Potentially inappropriate medical claim detected in response")
                response = "I appreciate you sharing with me. While I can offer support and coping strategies, I'm not qualified to provide medical diagnoses or treatment recommendations. I encourage you to speak with a qualified mental health professional who can properly assess and help you. Would you like me to suggest some coping techniques in the meantime?"
        
        return response
    

# 1. => b. ==================================================================================================== #  
    def _get_fallback_response(
        self,
        safety_context: Optional[Dict[str, Any]]
    ) -> str:
        """
        Generate a fallback response when LLM fails
        """
        if safety_context and safety_context.get("risk_level") in ["high", "critical"]:
            return """I'm having trouble processing your message right now, but I can see this might be urgent. 

                If you're experiencing a crisis or need immediate help:
                - Emergency Helpline: 1-800-273-8255
                - Text HOME to 741741 (Crisis Text Line)
                - Call 911 for emergencies

                Please reach out to a counselor or trusted person right away. Your wellbeing matters."""
        
        return """I'm having a brief technical difficulty, but I'm here to help. While I work on that, here are a few things that might help:

                - Take a few deep breaths
                - Try a grounding exercise (5 things you can see, 4 you can touch, 3 you can hear, 2 you can smell, 1 you can taste)
                - Consider talking to a counselor or trusted friend

                Please try sending your message again in a moment, or connect with a counselor directly if you need immediate support."""




# 2.  ====================================================================================================================================== #  
    async def generate_coping_strategies(
        self,
        concern_type: str,
        context: str = ""
    ) -> List[str]:
        """
        Generate specific coping strategies for a concern type
        """
        prompt = f"""Generate 3-5 practical, evidence-based coping strategies for someone experiencing {concern_type}.

            Context: {context}

            Provide specific, actionable strategies that a college student can easily implement.
            Focus on CBT techniques, mindfulness, and self-care practices."""

        try:
            # FIX: Use ainvoke for this call as well
            response = await self.llm.ainvoke([HumanMessage(content=prompt)])
            
            # Parse strategies from response
            strategies = [
                s.strip() 
                for s in response.content.split('\n') 
                if s.strip() and not s.strip().startswith('#')
            ]
            
            return strategies[:5]
            
        except Exception as e:
            logger.error(f"Strategy generation error: {str(e)}")
            return self._get_default_strategies(concern_type)
    

# 2. => a.   ==================================================================================================== #  
    def _get_default_strategies(self, concern_type: str) -> List[str]:
        """Default coping strategies by concern type"""
        strategies = {
            "anxiety": [
                "Practice deep breathing: Inhale for 4 counts, hold for 4, exhale for 4",
                "Try the 5-4-3-2-1 grounding technique",
                "Write down your worries and set aside 'worry time'",
                "Progressive muscle relaxation exercise"
            ],
            "stress": [
                "Break tasks into smaller, manageable steps",
                "Take regular breaks during study sessions",
                "Practice mindful walking or movement",
                "Maintain a consistent sleep schedule"
            ],
            "depression": [
                "Set one small achievable goal each day",
                "Spend time outdoors if possible",
                "Connect with a friend or family member",
                "Engage in activities you used to enjoy, even briefly"
            ]
        }
        
        return strategies.get(concern_type.lower(), [
            "Take a few deep breaths",
            "Talk to someone you trust",
            "Engage in a calming activity",
            "Consider reaching out to a counselor"
        ])
